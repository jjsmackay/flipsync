# Pinned <4.5: ctranslate2 4.5+ requires cuDNN 9, whereas 4.4.x needs cuDNN 8.
# faster-whisper 1.0.3 is compatible with this range. CUDA availability is
# detected via ctranslate2.get_cuda_device_count(); torch is no longer a
# dependency (it was imported solely for cuda.is_available()).
ctranslate2<4.5
# The ctranslate2 4.4.x wheel bundles ONLY the cuDNN front-end stub, not the
# modular sub-libraries (libcudnn_ops_infer.so.8, libcudnn_cnn_infer.so.8, ...)
# that get dlopen'd at the first GPU inference — so on a bare python:3.11-slim
# base the process dies mid-transcription with "Could not load library
# libcudnn_ops_infer.so.8". Install the full cuDNN 8 runtime (pulls
# nvidia-cublas-cu12 / nvidia-cuda-nvrtc-cu12 transitively) and put it on
# LD_LIBRARY_PATH via the Dockerfile so the lazy dlopen resolves.
nvidia-cudnn-cu12==8.9.7.29
